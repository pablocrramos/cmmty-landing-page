Diseño del componente <Section>

Revisa la estructura, el API de props y el comportamiento visual por breakpoint.

1 — API del componente

// src/components/atoms/Section.tsx

interface SectionProps {
  variant?: 'white' | 'light-gray'   // bg + color de línea automático
  id?: string                          // anchor links (#servicios, etc.)
  className?: string                   // clases extra para el <section>
  innerClassName?: string              // clases extra para el contenedor
  children: React.ReactNode
}

// Uso:
<Section variant="white">
  {/* contenido */}
</Section>

<Section variant="light-gray" id="servicios">
  {/* contenido */}
</Section>
